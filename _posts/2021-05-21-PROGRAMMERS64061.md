### 문제 풀이1

```kotlin
import java.util.Stack

class Solution {
    fun solution(board: Array<IntArray>, moves: IntArray): Int {
        val stack = Stack<Int>()
        var count = 0

        moves.forEach { x ->
            for (y in board.indices) {
                if (board[y][x - 1] != 0) {
                    if (stack.isEmpty()) {
                        stack.push(board[y][x - 1])
                    } else {
                        if (stack.peek() == board[y][x - 1]) {
                            stack.pop()
                            count += 2
                        } else {
                            stack.push(board[y][x - 1])
                        }
                    }

                    board[y][x - 1] = 0
                    break
                }
            }
        }

        return count
    }
}
```
